Step 1 — Fetch Job Openings (Adzuna API)
import requests, pandas as pd

APP_ID  = "your_adzuna_app_id"
APP_KEY = "your_adzuna_key"

def fetch_jobs(state, page=1):
    url = f"https://api.adzuna.com/v1/api/jobs/in/search/{page}"
    r = requests.get(url, params={
        "app_id":    APP_ID,
        "app_key":   APP_KEY,
        "where":     state,
        "results_per_page": 50,
        "what":      "jobs"
    })
    return r.json().get("results", [])

# Loop all states
all_jobs = []
for state in INDIA_STATES:
    jobs = fetch_jobs(state)
    for j in jobs:
        all_jobs.append({
            "title":    j["title"],
            "company":  j["company"]["display_name"],
            "location": j["location"]["display_name"],
            "state":    state,
            "salary":   j.get("salary_min", 0),
            "category": j["category"]["label"],
            "date":     j["created"]
        })

df_jobs = pd.DataFrame(all_jobs)
Step 2 — Load Census Migration Origin Data
# Download from:
# censusindia.gov.in → D-02 Tables (Migrants by last residence)
# OR microdata.gov.in → PLFS 2020-21

df_migration = pd.read_csv("census_d02_migrants.csv")

# Filter for "reason = work/employment"
df_work = df_migration[
    df_migration["reason_for_migration"]
    .isin(["Work/Employment", "Business"])
]

# Build origin → destination matrix
pivot = df_work.pivot_table(
    index="state_of_last_residence",
    columns="state_of_enumeration",
    values="persons",
    aggfunc="sum"
).fillna(0)
Step 3 — Merge & Build Flow Dataset
job_counts = (df_jobs
    .groupby("state")["title"]
    .count()
    .reset_index()
    .rename(columns={"title":"job_openings"})
)

# Migration flows with job destination weight
flows = []
for _, row in df_work.iterrows():
    dest = row["state_of_enumeration"]
    job_weight = job_counts.loc[
        job_counts.state == dest, "job_openings"
    ].values[0] if dest in job_counts.state.values else 0
    flows.append({
        "origin":      row["state_of_last_residence"],
        "destination": dest,
        "migrants":    row["persons"],
        "job_openings": job_weight,
        "flow_ratio":  row["persons"] / (job_weight + 1)
    })

df_flows = pd.DataFrame(flows)
df_flows.to_json("migration_flows.json", orient="records")
